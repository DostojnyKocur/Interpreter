program						:	statement
compound_statement			:	SCOPEBEGIN statement_list SCOPEEND
statement_list				:	statement | statement statement_list
statement					:	compound_statement | assignment_statement SEMI | variables_declarations | function_definition | function_call SEMI | return_statement SEMI | while_statement | if_statement | empty
assignment_statement		:	variable ASSIGN expression | variables_declarations ASSIGN expression
return_statement			:	RETURN expression 
while_statement				:	WHILE LPAREN condition RPAREN statement
if_statement				:	IF LPAREN condition RPAREN statement (ELSE statement)?
expression					:	term ((PLUS | MINUS) term)* | condition
term						:	factor ((MUL | DIV | MOD) factor)*
factor						:	(PLUS | MINUS) factor | NUMBER_CONST | LPAREN expression RPAREN | function_call
condition					: 	or_condition
or_condition				:	and_condition (OR and_condition)*
and_condition				:	not_condition (AND not_condition)*
not_condition				:	NOT not_condition | comparison
comparison					: 	expression (COMAPRE expression)* | LPAREN condition RPAREN
function_definition			:	type_specification ID LPAREN parameter_list+ RPAREN compound_statement
parameter_list				:	(type_specification ID (COMMA)+ )*
function_call				:	ID LPAREN (expression (COMMA expression)*)? RPAREN
variables_declarations		:	variable_declaration*
variable_declaration		:	type_specification ID (COMMA ID)* SEMI
variable					:	ID
type_specification			:	NUMBER
empty						:
