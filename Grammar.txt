program						:	statement
compound_statement			:	SCOPEBEGIN statement_list SCOPEEND
statement_list				:	statement 
								| statement statement_list
statement					:	compound_statement 
								| assignment_statement SEMI 
								| variables_declarations 
								| function_definition 
								| function_call SEMI 
								| return_statement SEMI 
								| while_statement 
								| for_statement
								| if_statement 
								| empty
assignment_statement		:	variable ASSIGN expression 
								| variables_declarations ASSIGN expression
								| variable ASSIGN array_initialization
								| variables_declarations ASSIGN array_initialization
return_statement			:	RETURN condition 
break_statement				:	BREAK SEMI
continue_statement			:	CONTINUE SEMI
while_statement				:	WHILE LPAREN condition RPAREN statement
for_statement				:	FOR LPAREN (assignment_statement (COMMA assignment_statement)*)? SEMI condition? SEMI (assignment_statement (COMMA assignment_statement)*)? RPAREN statement
if_statement				:	IF LPAREN condition RPAREN statement (ELSE statement)?
array_initialization		:	LBRACKET (expression (COMMA expression)*)? RBRACKET
index_expression			:	variable LBRACKET expression RBRACKET
expression					:	term ((PLUS | MINUS) term)* 
								| condition
term						:	factor ((MUL | DIV | MOD) factor)*
factor						:	(PLUS | MINUS) factor 
								| LPAREN expression RPAREN 
								| function_call
								| index_expression
								| NUMBER_CONST 
								| BOOL_CONST
								| STRING_CONST
condition					: 	or_condition
or_condition				:	and_condition (OR and_condition)*
and_condition				:	not_condition (AND not_condition)*
not_condition				:	NOT not_condition 
								| LPAREN condition RPAREN 
								| comparison
comparison					: 	expression (COMAPRE expression)* 
								| TRUE 
								| FALSE
function_definition			:	type_specification ID LPAREN parameter_list+ RPAREN compound_statement
parameter_list				:	(type_specification ID (COMMA)+ )*
function_call				:	ID LPAREN (expression (COMMA expression)*)? RPAREN
variables_declarations		:	variable_declaration*
variable_declaration		:	type_specification ID (COMMA ID)* SEMI
variable					:	ID
								| ID LBRACKET expression RBRACKET
non_array_specification		:	VOID
								| NUMBER
								| BOOL
								| STRING
rank_specifier				:	LBRACKET RBRACKET
array_specification			:	non_array_specification rank_specifier+
type_specification			:	non_array_specification
								| array_specification
empty						:
